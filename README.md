# portscan.py

Simple python port scanner

# usage

python portscan.py -H hostname -p ports (seperated by commas, eg: -p 21,22,80)

# requirements

pyFiglet

# installation

git clone https://github.com/norksec/portscan.git

pip install pyfiglet
